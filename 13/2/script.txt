const renderProducts = (maxPrice) => {
  const productWrapper = document.querySelector('.product_wrapper')
  productWrapper.innerHTML = ''
  const filteredProducts = products.filter(
    (product) => product.price <= maxPrice
  )
  filteredProducts.forEach((product) => {
    const productDiv = document.createElement('div')
    productDiv.className = 'product'
    productDiv.innerHTML = `
      <h2>${product.name}</h2>
      <p>${product.description}</p>
      <p>Price: $${product.price}</p>
    `
    productWrapper.appendChild(productDiv)
  })
}

document.getElementById('priceInput').addEventListener('input', (event) => {
  const maxPrice = parseFloat(event.target.value) || Infinity
  renderProducts(maxPrice)
})

renderProducts(Infinity)
